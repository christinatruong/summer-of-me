<script src="http://code.jquery.com/jquery-1.11.0.min.js"></script>
<script>
	$(function() {

		$("button").on("click", function(){
			$("img").removeClass();
			var filter = $(this).data("filter");
			$("img").addClass(filter);
		});

	});
</script>
